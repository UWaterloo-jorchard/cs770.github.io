Jacobi Step¶

In [20]:
def Jacobi(A,b,xold):
    '''
     xnew = Jacobi(A, b, xold)
     
     Takes one Jacobi step for a 2x2 system.
    '''
    xnew = np.zeros_like(b)
    xnew[0] = (b[0] - A[0,1]*xold[1]) / A[0,0]
    xnew[1] = (b[1] - A[1,0]*xold[0]) / A[1,1]
    return xnew
